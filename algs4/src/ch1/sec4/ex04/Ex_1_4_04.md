class TwoSum {
    public static int count(int[] a) {
        int count = 0;                          // A
        for (int i = 0; i < n; i++) {           // B
            for (int j = i + 1; j < n; j++) {   // C - including `if`
                if (a[i] + a[j] == 0) {
                    count++;                    // D
                }
            }
        }
        return count;
    }
}
